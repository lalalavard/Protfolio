## 快速开始

### 安装

```bash
# 克隆项目
git clone `repoUrl`

# 安装依赖
npm install

# 启动服务
npm run start

# 打包构建
npm run build
```

### 项目结构:

```
├───node_modules
├───pages
│   └───api
├───public
└───styles
```

### 使用方法

```jsx
import type { NextPage } from "next";
import Head from "next/head";
import Layout from "../components/layout";

const Projects: NextPage = ({ projects }: any) => {
  return <Layout>To be done...</Layout>;
};

export default Projects;
```
